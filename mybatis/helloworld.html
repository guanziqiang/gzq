//1，获取mybatis配置文件
    /*
        //使用类加载器获取mybatis的配置文件
    InputStream inputStream = 
       this.getClass().getClassLoader().getResourceAsStream("mybits.xml");
    */
    //使用mybatis提供的Resource获取配置文件
    Reader reader = Resources.getResourceAsReader("mybits.xml");
//2，创建session工厂
    SqlSessionFactory sqlSessionFactory = 
       new SqlSessionFactoryBuilder().build(reader);
//3，获取session，sqlSessionFactory.openSession(true)取值为true时可以自动提交。
    SqlSession sqlSession = sqlSessionFactory.openSession();
//4，使用session进行CRUD
    User user = sqlSession.selectOne("userxml."+"getUserByName", "TOM");
    System.out.println(user);
    //5，手动提交和关闭session。
    sqlSession.commit();
    sqlSession.close();
